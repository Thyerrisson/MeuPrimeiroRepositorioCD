ex1 exercícios de fixação e aprendizado

num1 = int(input("Digite um número: "))
num2 = int(input("Digite outro número: "))


if num1 <= num2 or num1 >= num2:
    soma = num1 + num2
    produto = num1 * num2
    diferenca = num2 - num1
    divisao = num2 / num1
    divisao_inteira = num2 // num1
    resto = num2 % num1
else:
    diferenca = num1 - num2
    divisao = num1 / num2
    divisao_inteira = num1 // num2
    resto = num1 % num2

print(f"A soma desses números é: {soma}")
print(f"A diferença desses números é de: {diferenca}")
print(f"O produto desses números é de: {produto}")
print(f"A divisão desses números é de: {divisao}")
print(f"A divisão inteira desses números é de: {divisao_inteira}")
print(f"O resto desses números é de: {resto}")
