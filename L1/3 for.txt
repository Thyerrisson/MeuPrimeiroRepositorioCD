#3. Estrutura de Repetição FOR:
#Exercício: Peça um número inteiro ao usuário e exiba a tabuada de
#multiplicação desse número de 1 a 10 utilizando o laço for e a
#função range().

x = range(1, 11)
taboada = 0

num = int(input("Digite um número inteiro: "))

for _ in x:
    taboada += 1
    x = num * taboada
    print(x)
