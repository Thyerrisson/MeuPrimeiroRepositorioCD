print("TechInvest: Sistema Prime Ativado")

nome = input("Qual o seu nome? ")
renda = float(input("Qual sua renda mensal? R$ "))
gastos = float(input("Qual seu gasto mensal total? R$ "))
coragem = int(input("De 1 a 10, qual seu nível de coragem para investir? "))

sobra = renda - gastos

print(f"Relatório de {nome}")

if gastos > renda:
    print("ALERTA: Emergência Financeira! Você está gastando mais do que ganha.")
else:
    reserva_ideal = gastos * 6
    print(f"Parabéns! Sobram R$ {sobra} por mês.")
    if sobra < reserva_ideal:
        falta = reserva_ideal - sobra
        print(f"Atenção: Para sua reserva de segurança (6 meses), ainda faltam R$ {falta}.")

print("Perfil de Investimento")
if coragem < 4:
    perfil = "Medroso"
    recomendacao = "Tesouro Direto"
elif 4 <= coragem <= 7:
    perfil = "Cauteloso"
    recomendacao = "Fundos Imobiliários"
else:
    perfil = "Corajoso!"
    recomendacao = "Ações de Tecnologia"

print(f"O seu perfil é: {perfil}, e a sua recomendação é: {recomendacao}")

print("Simulação de Futuro")
anos = int(input("Quantos anos você pretende investir? "))
meses = anos * 12
valor_acumulado = 0

for mes in range(1, meses + 1):
    valor_acumulado += sobra
    valor_acumulado *= 1.005 

print(f"Em {anos} anos, investindo sua sobra mensal com disciplina,")
print(f"você teria aproximadamente: R$ {valor_acumulado}")
print("==========================================")
